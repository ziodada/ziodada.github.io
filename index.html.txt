    <script>
        const prizes = [
            { name: "Sconto del 10%", probability: 0.4 },
            { name: "Sessione gratuita", probability: 0.3 },
            { name: "Buono regalo da 50‚Ç¨", probability: 0.2 },
            { name: "Pacchetto Deluxe", probability: 0.1 }
        ];

        let selectedPrize;

        function pickGift() {
            const random = Math.random();
            let cumulativeProbability = 0;
            
            for (let prize of prizes) {
                cumulativeProbability += prize.probability;
                if (random <= cumulativeProbability) {
                    selectedPrize = prize;
                    break;
                }
            }

            document.getElementById('giftPicker').style.display = 'none';
            document.getElementById('giftReveal').style.display = 'block';
        }

        function revealGift() {
            document.getElementById('giftReveal').style.display = 'none';
            document.getElementById('giftResult').style.display = 'block';
            document.getElementById('prize').textContent = selectedPrize.name;
        }

        function reset() {
            document.getElementById('giftResult').style.display = 'none';
            document.getElementById('giftPicker').style.display = 'block';
        }
    </script>
